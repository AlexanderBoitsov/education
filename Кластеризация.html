Проведём кластеризацию методом Mean shift, т.к. в исходных данных не должно быть выбросов, а форма кластеров заранее неизвестна

In [9]:
params = np.arange(0.01, 0.21, 0.01)
num_clusters = []
sil_score = []
bandwidths = []

for quantile in params:
    bandwidth = estimate_bandwidth(data_for_classification, quantile=quantile)
    mean_shift = MeanShift(bandwidth=bandwidth).fit(data_for_classification)
    bandwidths.append(bandwidth)
    num_clusters.append(len(np.unique(mean_shift.labels_)))
    if len(np.unique(mean_shift.labels_)) == 1:
        sil_score.append(0.0)
    else:
        sil_score.append(silhouette_score(data_for_classification, mean_shift.labels_))

fig, ax1 = plt.subplots()

color='tab:red'
ax1.set_xlabel('Полоса пропускания')
ax1.set_ylabel('Число кластеров', color='tab:red')
ax1.plot(bandwidths, num_clusters, color=color)
ax1.tick_params(axis='y', labelcolor=color)

ax2 = ax1.twinx()  

color = 'tab:green'
ax2.set_ylabel('Коэффициент силуэта', color=color) 
ax2.plot(bandwidths, sil_score, color=color)
ax2.tick_params(axis='y', labelcolor=color)

fig.tight_layout() 
plt.show()
