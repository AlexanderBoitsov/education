Age_and_Credit_Limit = data[["Customer_Age", "Credit_Limit"]].sort_values(by = ["Customer_Age"])
Credit_Limit = Age_and_Credit_Limit["Credit_Limit"]
Customer_Age = Age_and_Credit_Limit["Customer_Age"]

N = 19
Age = []
Cred_Lim = []
step = int(len(Age_and_Credit_Limit)/N)
for i in range(N):
    start = i*step
    stop = (i+1)*step
    Age.append(np.mean(Age_and_Credit_Limit["Customer_Age"][start:stop]))
    Cred_Lim.append(np.mean(Age_and_Credit_Limit["Credit_Limit"][start:stop]))

Age_new = np.linspace(30, 60, 180)

plt.plot(Age_new, PchipInterpolator(Age, Cred_Lim)(Age_new), color = 'r')
plt.plot(Age, Cred_Lim, 'o')
plt.legend(fontsize=12)
plt.grid(which='major')
plt.xlim([30, 60])
plt.xlabel(r'Возраст', fontsize=12)
plt.ylabel(r'Усреднённый лимит кредита', fontsize=12)
text_for_boxplot = """
Клиенты от 40 до 52 лет демонстрируют наибольшее 
      усреднённое значение по лимиту их кредита

"""
plt.text(33, 5250, text_for_boxplot)
pdf.savefig(bbox_inches = "tight")
